<script>
	function getHtml(file){
    		return new Promise((resolve) => {
      			fetch(file)
        			.then((response) => {
         				return response.text();
        			})
        		.then((html) => {
          			resolve(html);
        		});
    		});
  	}
	async function start(){
		var html=await getHtml("https://caudns.vercel.app/data.txt");
		html=html.toString();
		console.log(html);
		document.body.innerHTML=html;
	}
	start();
</script>
